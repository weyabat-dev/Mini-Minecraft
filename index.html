const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); let player = { x:350, y:220, size:20, health:100 }; let bats = [ {x:100,y:100,size:15}, {x:600,y:80,size:15}, {x:200,y:400,size:15} ]; let vampire = { x:50, y:300, size:25, health:50 }; let attack = null; document.addEventListener("keydown",function(e){ if(e.key=="ArrowUp") player.y-=10; if(e.key=="ArrowDown") player.y+=10; if(e.key=="ArrowLeft") player.x-=10; if(e.key=="ArrowRight") player.x+=10; if(e.key==" "){ attack={ x:player.x, y:player.y, size:15, life:10 }; } }); function drawPlayer(){ ctx.fillStyle="lime"; ctx.fillRect(player.x,player.y,player.size,player.size); } function drawBats(){ ctx.fillStyle="purple"; bats.forEach(b=>{ ctx.fillRect(b.x,b.y,b.size,b.size); }); } function drawVampire(){ ctx.fillStyle="red"; ctx.fillRect(vampire.x,vampire.y,vampire.size,vampire.size); } function moveEnemies(){ bats.forEach(b=>{ if(b.xplayer.x) b.x-=1; if(b.yplayer.y) b.y-=1; }); if(vampire.xplayer.x) vampire.x-=0.5; if(vampire.yplayer.y) vampire.y-=0.5; } function drawAttack(){ if(!attack) return; ctx.fillStyle="yellow"; ctx.fillRect(attack.x,attack.y,attack.size,attack.size); attack.life--; if(attack.life<=0) attack=null; } function checkCollisions(){ bats.forEach((b,i)=>{ if(Math.abs(player.x-b.x)<20 && Math.abs(player.y-b.y)<20){ player.health-=1; } if(attack && Math.abs(attack.x-b.x)<20 && Math.abs(attack.y-b.y)<20){ bats.splice(i,1); } }); if(Math.abs(player.x-vampire.x)<25 && Math.abs(player.y-vampire.y)<25){ player.health-=2; } if(attack && Math.abs(attack.x-vampire.x)<25 && Math.abs(attack.y-vampire.y)<25){ vampire.health-=5; } if(vampire.health<=0){ vampire.x=-100; } } function updateUI(){ document.getElementById("health").textContent=player.health; } function game(){ ctx.clearRect(0,0,700,450); drawPlayer(); drawBats(); drawVampire(); drawAttack(); moveEnemies(); checkCollisions(); updateUI(); requestAnimationFrame(game); } game();
